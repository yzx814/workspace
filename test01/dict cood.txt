print("========字典操作=======")
a = {"name": "z", "sex": "men", "age": "36"}  #字典的一般形式
print(len(a))   #len 函数是指字典的长度
print(a["name"])  #字典中所对应的值
a["age"] = 20    #修改年龄
print(a)
a["height"]=1.72    #增添身高
print(a)

a.pop("sex")   #pop删除属性
